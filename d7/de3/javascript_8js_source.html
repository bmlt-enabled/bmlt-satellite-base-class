39  extra_data ///< If supplied, extra data to be delivered to the callback.

42  /************************************************************************************//**
43  * \brief Create a generic XMLHTTPObject. *
44  * *
45  * This will account for the various flavors imposed by different browsers. *
46  * *
47  * \returns a new XMLHTTPRequest object. *

50  function createXMLHTTPObject()
51  {
52  var XMLHttpArray = [
53  function() {return new XMLHttpRequest()},
54  function() {return new ActiveXObject("Msxml2.XMLHTTP")},
55  function() {return new ActiveXObject("Msxml2.XMLHTTP")},
56  function() {return new ActiveXObject("Microsoft.XMLHTTP")}
57  ];
58 
59  var xmlhttp = false;
60 
61  for ( var i=0; i < XMLHttpArray.length; i++ )
62  {
63  try
64  {
65  xmlhttp = XMLHttpArray[i]();
66  }
67  catch(e)
68  {
69  continue;
70  };
71  break;
72  };
73 
74  return xmlhttp;
75  };
76 
77  var req = createXMLHTTPObject();
78  req.finalCallback = callback;
79  var sVars = null;
80  method = method.toString().toUpperCase();
81  var drupal_kludge = '';
82 
83  // Split the URL up, if this is a POST.
84  if ( method == "POST" )
85  {
86  var rmatch = /^([^\?]*)\?(.*)$/.exec ( url );
87  url = rmatch[1];
88  sVars = rmatch[2];
89  // This horrible, horrible kludge, is because Drupal insists on having its q parameter in the GET list only.
90  var rmatch_kludge = /(q=admin\/settings\/bmlt)&?(.*)/.exec ( rmatch[2] );
91  if ( rmatch_kludge && rmatch_kludge[1] )
92  {
93  url += '?'+rmatch_kludge[1];
94  sVars = rmatch_kludge[2];
95  };
96  };
97  if ( extra_data )
98  {
99  req.extra_data = extra_data;
100  };
101  req.open ( method, url, true );
102  if ( method == "POST" )
103  {
104  req.setRequestHeader("Method", "POST "+url+" HTTP/1.1");
105  req.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
106  };
107  req.onreadystatechange = function ( )
108  {
109  if ( req.readyState != 4 ) return;
110  if( req.status != 200 ) return;
111  callback ( req, req.extra_data );
112  req = null;
113  };
114  req.send ( sVars );
115 
116  return req;
117 };

207 function sprintf() {
208  var i = 0, a, f = arguments[i++], o = [], m, p, c, x, s = '';
209  while (f) {
210  if (m = /^[^\x25]+/.exec(f)) {
211  o.push(m[0]);
212  }
213  else if (m = /^\x25{2}/.exec(f)) {
214  o.push('%');
215  }
216  else if (m = /^\x25(?:(\d+)\$)?(\+)?(0|'[^$])?(-)?(\d+)?(?:\.(\d+))?([b-fosuxX])/.exec(f)) {
217  if (((a = arguments[m[1] || i++]) == null) || (a == undefined)) {
218  throw('Too few arguments.');
219  };
220  if (/[^s]/.test(m[7]) && (typeof(a) != 'number')) {
221  throw('Expecting number but found ' + typeof(a));
222  };
223  switch (m[7]) {
224  case 'b': a = a.toString(2); break;
225  case 'c': a = String.fromCharCode(a); break;
226  case 'd': a = parseInt(a,10); break;
227  case 'e': a = m[6] ? a.toExponential(m[6]) : a.toExponential(); break;
228  case 'f': a = m[6] ? parseFloat(a).toFixed(m[6]) : parseFloat(a); break;
229  case 'o': a = a.toString(8); break;
230  case 's': a = ((a = String(a)) && m[6] ? a.substring(0, m[6]) : a); break;
231  case 'u': a = Math.abs(a); break;
232  case 'x': a = a.toString(16); break;
233  case 'X': a = a.toString(16).toUpperCase(); break;
234  };
235  a = (/[def]/.test(m[7]) && m[2] && a >= 0 ? '+'+ a : a);
236  c = m[3] ? m[3] == '0' ? '0' : m[3].charAt(1) : ' ';
237  x = m[5] - String(a).length - s.length;
238  p = m[5] ? str_repeat(c, x) : '';
239  o.push(s + (m[4] ? a + p : p + a));
240  }
241  else {
242  throw('Huh ?!');
243  };
244  f = f.substring(m[0].length);
245  };
246  return o.join('');
247 };
