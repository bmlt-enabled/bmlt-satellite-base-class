36function BMLTPlugin_AjaxRequest( url, ///< The URI to be called
37 callback, ///< The success callback
38 method, ///< The method ('get' or 'post')
39 extra_data ///< If supplied, extra data to be delivered to the callback.
40) {
41 /************************************************************************************//**
42 * \brief Create a generic XMLHTTPObject. *
43 * *
44 * This will account for the various flavors imposed by different browsers. *
45 * *
46 * \returns a new XMLHTTPRequest object. *
47 ****************************************************************************************/
48
49 function createXMLHTTPObject()
50 {
51 var XMLHttpArray = [
52 function () {
53 return new XMLHttpRequest()},
54 function () {
55 return new ActiveXObject("Msxml2.XMLHTTP")},
56 function () {
57 return new ActiveXObject("Msxml2.XMLHTTP")},
58 function () {
59 return new ActiveXObject("Microsoft.XMLHTTP")}
60 ];
61
62 var xmlhttp = false;
63
64 for ( var i=0; i < XMLHttpArray.length; i++ )
65 {
66 try
67 {
68 xmlhttp = XMLHttpArray[i]();
69 }
70 catch(e)
71 {
72 continue;
73 };
74 break;
75 };
76
77 return xmlhttp;
78 };
79
80 var req = createXMLHTTPObject();
81 req.finalCallback = callback;
82 var sVars = null;
83 method = method.toString().toUpperCase();
84 var drupal_kludge = '';
85
86 // Split the URL up, if this is a POST.
87 if (method == "POST" ) {
88 var rmatch = /^([^\?]*)\?(.*)$/.exec(url);
89 url = rmatch[1];
90 sVars = rmatch[2];
91 // This horrible, horrible kludge, is because Drupal insists on having its q parameter in the GET list only.
92 var rmatch_kludge = /(q=admin\/settings\/bmlt)&?(.*)/.exec(rmatch[2]);
93 if (rmatch_kludge && rmatch_kludge[1] ) {
94 url += '?'+rmatch_kludge[1];
95 sVars = rmatch_kludge[2];
96 };
97 };
98 if (extra_data ) {
99 req.extra_data = extra_data;
100 };
101 req.open(method, url, true);
102 if (method == "POST" ) {
103 req.setRequestHeader("Method", "POST "+url+" HTTP/1.1");
104 req.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
105 };
106 req.onreadystatechange = function ( ) {
107 if (req.readyState != 4 ) { return;
108 }
109 if(req.status != 200 ) { return;
110 }
111 callback(req, req.extra_data);
112 req = null;
113 };
114 req.send(sVars);
115
116 return req;
117};

205function sprintf()
206{
207 var i = 0, a, f = arguments[i++], o = [], m, p, c, x, s = '';
208 while (f) {
209 if (m = /^[^\x25]+/.exec(f)) {
210 o.push(m[0]);
211 }
212 else if (m = /^\x25{2}/.exec(f)) {
213 o.push('%');
214 }
215 else if (m = /^\x25(?:(\d+)\$)?(\+)?(0|'[^$])?(-)?(\d+)?(?:\.(\d+))?([b-fosuxX])/.exec(f)) {
216 if (((a = arguments[m[1] || i++]) == null) || (a == undefined)) {
217 throw('Too few arguments.');
218 };
219 if (/[^s]/.test(m[7]) && (typeof(a) != 'number')) {
220 throw('Expecting number but found ' + typeof(a));
221 };
222 switch (m[7]) {
223 case 'b': a = a.toString(2); break;
224 case 'c': a = String.fromCharCode(a); break;
225 case 'd': a = parseInt(a,10); break;
226 case 'e': a = m[6] ? a.toExponential(m[6]) : a.toExponential(); break;
227 case 'f': a = m[6] ? parseFloat(a).toFixed(m[6]) : parseFloat(a); break;
228 case 'o': a = a.toString(8); break;
229 case 's': a = ((a = String(a)) && m[6] ? a.substring(0, m[6]) : a); break;
230 case 'u': a = Math.abs(a); break;
231 case 'x': a = a.toString(16); break;
232 case 'X': a = a.toString(16).toUpperCase(); break;
233 };
234 a = (/[def]/.test(m[7]) && m[2] && a >= 0 ? '+'+ a : a);
235 c = m[3] ? m[3] == '0' ? '0' : m[3].charAt(1) : ' ';
236 x = m[5] - String(a).length - s.length;
237 p = m[5] ? str_repeat(c, x) : '';
238 o.push(s + (m[4] ? a + p : p + a));
239 }
240 else {
241 throw('Huh ?!');
242 };
243 f = f.substring(m[0].length);
244 };
245 return o.join('');
246};
