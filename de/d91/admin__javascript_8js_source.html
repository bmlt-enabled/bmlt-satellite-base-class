49/****************************************************************************************//**
50* \brief Hides "primer" text in text items. *
51********************************************************************************************/
52function BMLTPlugin_ClickInText ( in_id, ///< The ID of the item
53 in_default_text, ///< The "primer" text.
54 in_blur ///< If this is true, then we reverse the process.
55 )
56{
57 var item = document.getElementById(in_id);
58
59 if ( item && (item.type == 'text') )
60 {
61 var value = item.value.toString();
62
63 if ( (in_blur != true) && value && (value == in_default_text) )
64 {
65 item.value = '';
66 }
67 else
68 {
69 if ( (in_blur == true) && !value )
70 {
71 item.value = in_default_text;
72 };
73 };
74 };
75}

274 )
275{

316/****************************************************************************************//**
317* \brief When a change is made to an option, this sets a "dirty" flag, by enabling the *
318* "Save Changes" button. *
319********************************************************************************************/
320function BMLTPlugin_DirtifyOptionSheet( in_disable ///< If this is true, then we "clean" the flag.
321 )
322{
323 var keyID = (window.event && window.event.keyCode) ? window.event.keyCode : null;
324
325 if ( keyID != 9 ) // Don't react to tab.
326 {
327 document.getElementById ( 'BMLTPlugin_toolbar_button_save' ).disabled = (in_disable == true);
328 document.getElementById ( 'BMLTPlugin_toolbar_button_save' ).className = 'BMLTPlugin_toolbar_button_save_' + ((in_disable == true) ? 'disabled' : 'enabled').toString();
329
330 document.getElementById ( 'BMLTPlugin_toolbar_button_new' ).disabled = (in_disable != true);
331 document.getElementById ( 'BMLTPlugin_toolbar_button_new' ).className = ((in_disable != true) ? 'BMLTPlugin_toolbar_button_save_disabled' : 'BMLTPlugin_create_button').toString();
332
333 if ( document.getElementById ( 'BMLTPlugin_toolbar_button_del' ) )
334 {
335 document.getElementById ( 'BMLTPlugin_toolbar_button_del' ).disabled = (in_disable != true);
336 document.getElementById ( 'BMLTPlugin_toolbar_button_del' ).className = ((in_disable != true) ? 'BMLTPlugin_toolbar_button_save_disabled' : 'BMLTPlugin_delete_button').toString();
337 };
338
339 if ( !document.getElementById ( 'BMLTPlugin_toolbar_button_save' ).disabled )
340 {
341 if ( window.onbeforeunload != BMLTPlugin_CloseHandler )
342 {
343 g_BMLTPlugin_oldBeforeUnload = window.onbeforeunload;
344 window.onbeforeunload = BMLTPlugin_CloseHandler;
345 };
346 }
347 else
348 {
349 window.onbeforeunload = g_BMLTPlugin_oldBeforeUnload;
350 };
351 };

363/****************************************************************************************//**
364* \brief Starts the message "fader." *
365* *
366* Simple fader, taken from here: *
367* http://www.switchonthecode.com/tutorials/javascript-tutorial-simple-fade-animation *
368********************************************************************************************/
369function BMLTPlugin_StartFader()
370{
371 var eid = 'BMLTPlugin_Fader';
372 var element = document.getElementById ( eid );
373
374 if ( element )
375 {
376 if ( element.FadeState == null )
377 {
378 if ( element.style.opacity == null
379 || element.style.opacity == ''
380 || element.style.opacity == '1' )
381 {
382 element.FadeState = 2;
383 }
384 else
385 {
386 element.FadeState = -2;
387 };
388 };
389
390 if ( element.FadeState == 1 || element.FadeState == -1 )
391 {
392 element.FadeState = element.FadeState == 1 ? -1 : 1;
393 element.FadeTimeLeft = g_BMLTPlugin_TimeToFade - element.FadeTimeLeft;
394 }
395 else
396 {
397 element.FadeState = element.FadeState == 2 ? -1 : 1;
398 element.FadeTimeLeft = g_BMLTPlugin_TimeToFade;
399 setTimeout ( "BMLTPlugin_animateFade(" + new Date().getTime() + ",'" + eid + "')", 33);
400 };
401 };
402};

404/****************************************************************************************//**
405* \brief Animates the fade. *
406* *
407* Simple fader, taken from here: *
408* http://www.switchonthecode.com/tutorials/javascript-tutorial-simple-fade-animation *
409********************************************************************************************/
410function BMLTPlugin_animateFade ( lastTick, ///< The time of the last tick.
411 eid ///< The element ID
412 )
413{
414 var curTick = new Date().getTime();
415 var elapsedTicks = curTick - lastTick;
416
417 var element = document.getElementById ( eid );
418
419 if ( element.FadeTimeLeft <= elapsedTicks )
420 {
421 element.style.opacity = element.FadeState == 1 ? '1' : '0';
422 element.style.filter = 'alpha(opacity = ' + (element.FadeState == 1 ? '100' : '0') + ')';
423 element.FadeState = element.FadeState == 1 ? 2 : -2;
424 return;
425 };
426
427 element.FadeTimeLeft -= elapsedTicks;
428
429 var newOpVal = element.FadeTimeLeft/g_BMLTPlugin_TimeToFade;
430
431 if ( element.FadeState == 1 )
432 {
433 newOpVal = 1 - newOpVal;
434 };
435
436 element.style.opacity = newOpVal;
437 element.style.filter = 'alpha(opacity = ' + (newOpVal*100) + ')';
438
439 setTimeout ( "BMLTPlugin_animateFade(" + curTick + ",'" + eid + "')", 33 );
440};
