363 /****************************************************************************************//**
364 * \brief Starts the message "fader." *
365 * *
366 * Simple fader, taken from here: *
367 * http://www.switchonthecode.com/tutorials/javascript-tutorial-simple-fade-animation *
368 ********************************************************************************************/
369 function BMLTPlugin_StartFader()
370 {
371  var eid = 'BMLTPlugin_Fader';
372  var element = document.getElementById ( eid );
373 
374  if ( element )
375  {
376  if ( element.FadeState == null )
377  {
378  if ( element.style.opacity == null
379  || element.style.opacity == ''
380  || element.style.opacity == '1' )
381  {
382  element.FadeState = 2;
383  }
384  else
385  {
386  element.FadeState = -2;
387  };
388  };
389 
390  if ( element.FadeState == 1 || element.FadeState == -1 )
391  {
392  element.FadeState = element.FadeState == 1 ? -1 : 1;
393  element.FadeTimeLeft = g_BMLTPlugin_TimeToFade - element.FadeTimeLeft;
394  }
395  else
396  {
397  element.FadeState = element.FadeState == 2 ? -1 : 1;
398  element.FadeTimeLeft = g_BMLTPlugin_TimeToFade;
399  setTimeout ( "BMLTPlugin_animateFade(" + new Date().getTime() + ",'" + eid + "')", 33);
400  };
401  };

404 /****************************************************************************************//**
405 * \brief Animates the fade. *
406 * *
407 * Simple fader, taken from here: *
408 * http://www.switchonthecode.com/tutorials/javascript-tutorial-simple-fade-animation *
409 ********************************************************************************************/
410 function BMLTPlugin_animateFade ( lastTick, ///< The time of the last tick.
411  eid ///< The element ID
412  )
413 {
414  var curTick = new Date().getTime();
415  var elapsedTicks = curTick - lastTick;
416 
417  var element = document.getElementById ( eid );
418 
419  if ( element.FadeTimeLeft <= elapsedTicks )
420  {
421  element.style.opacity = element.FadeState == 1 ? '1' : '0';
422  element.style.filter = 'alpha(opacity = ' + (element.FadeState == 1 ? '100' : '0') + ')';
423  element.FadeState = element.FadeState == 1 ? 2 : -2;
424  return;
425  };
426 
427  element.FadeTimeLeft -= elapsedTicks;
428 
429  var newOpVal = element.FadeTimeLeft/g_BMLTPlugin_TimeToFade;
430 
431  if ( element.FadeState == 1 )
432  {
433  newOpVal = 1 - newOpVal;
434  };
435 
436  element.style.opacity = newOpVal;
437  element.style.filter = 'alpha(opacity = ' + (newOpVal*100) + ')';
438 
439  setTimeout ( "BMLTPlugin_animateFade(" + curTick + ",'" + eid + "')", 33 );
440 };
