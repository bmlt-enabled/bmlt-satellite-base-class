366  /************************************************************************************//**
367  * \brief Determine the zoom level necessary to show all the markers in the viewport. *
368  ****************************************************************************************/
369 
370  function fit_markers()
371  {
372  var bounds = new google.maps.LatLngBounds();
373 
374  // We go through all the results, and get the "spread" from them.
375  for ( var c = 0; c < g_main_map.response_object.length; c++ )
376  {
377  var lat = g_main_map.response_object[c].latitude;
378  var lng = g_main_map.response_object[c].longitude;
379  // We will set our minimum and maximum bounds.
380  bounds.extend ( new google.maps.LatLng ( lat, lng ) );
381  };
382 
383  bounds.extend ( g_main_map.g_location_coords );
384 
385  // We now have the full rectangle of our meeting search results. Scale the map to fit them.
386  g_main_map.fitBounds ( bounds );
387  };
388 
389  /************************************************************************************//**
390  * \brief This calculates a circle that will fit all of the meetings found. *
391  * *
392  * \returns A boolean. True if the circle was successfully calculated. *
393  ****************************************************************************************/
394  function fit_circle()
395  {
396  var ret = false;
397  if ( g_main_map._circle_overlay )
398  {
399  var search_radius = g_search_radius;
400 
401  if ( !search_radius )
402  {
403  search_radius = 0.0;
404  // What we do here, is look at each meeting, and determine a circle that just fits them all.
405  for ( var c = 0; c < g_main_map.response_object.length; c++ )
406  {
407  var meeting_object = g_main_map.response_object[c];
408  var distance = google.maps.geometry.spherical.computeDistanceBetween ( g_main_map.g_location_coords, new google.maps.LatLng(meeting_object.latitude, meeting_object.longitude) );
409  search_radius = Math.max ( distance, search_radius );
410  };
411 
412  eval ( 'var dist_r_km = c_g_distance_units_are_km_'+g_main_id+';' );
413  // We only allow a maximum of 150% the highest value, so we don't have ginormous circles.
414  if ( search_radius > (c_g_diameter_choices[c_g_diameter_choices.length-1] * (dist_r_km ? 2000 : 2414.016)) )
415  {
416  search_radius = null;
417  };
418  };
419 
420  if ( search_radius )
421  {
422  g_search_radius = search_radius;
423  g_main_map._circle_overlay.setRadius ( g_search_radius );
424  g_main_map._circle_overlay.setCenter ( g_main_map.g_location_coords );
425  g_main_map.geo_width = g_search_radius / 1000.0;
426  ret = true;
427  };
428  };
429 
430  return ret;
431  };

515  /************************************************************************************//**
516  * \brief This returns an array, mapping out markers that overlap. *
517  * *
518  * \returns An array of arrays. Each array element is an array with n >= 1 elements, *
519  * each of which is a meeting object. Each of the array elements corresponds to a *
520  * single marker, and all the objects in that element's array will be covered by that *
521  * one marker. The returned sub-arrays will be sorted in order of ascending weekday. *
522  ****************************************************************************************/
523 
524  function mapOverlappingMarkers (in_meeting_array ///< Used to draw the markers when done.
525  )
526  {
527  var tolerance = 8; /* This is how many pixels we allow. */
528  var tmp = new Array;
529 
530  for ( var c = 0; c < in_meeting_array.length; c++ )
531  {
532  tmp[c] = new Object;
533  tmp[c].matched = false;
534  tmp[c].matches = null;
535  tmp[c].object = in_meeting_array[c];
536  tmp[c].coords = fromLatLngToPixel ( new google.maps.LatLng ( tmp[c].object.latitude, tmp[c].object.longitude ), g_main_map );
537  };
538  for ( var c = 0; c < in_meeting_array.length; c++ )
539  {
540  if ( false == tmp[c].matched )
541  {
542  tmp[c].matched = true;
543  tmp[c].matches = new Array;
544  tmp[c].matches[0] = tmp[c].object;
545 
546  for ( var c2 = 0; c2 < in_meeting_array.length; c2++ )
547  {
548  if ( false == tmp[c2].matched )
549  {
550  var outer_coords = tmp[c].coords;
551  var inner_coords = tmp[c2].coords;
552 
553  var xmin = outer_coords.x - tolerance;
554  var xmax = outer_coords.x + tolerance;
555  var ymin = outer_coords.y - tolerance;
556  var ymax = outer_coords.y + tolerance;
557 
558  /* We have an overlap. */
559  if ( (inner_coords.x >= xmin) && (inner_coords.x <= xmax) && (inner_coords.y >= ymin) && (inner_coords.y <= ymax) )
560  {
561  tmp[c].matches[tmp[c].matches.length] = tmp[c2].object;
562  tmp[c2].matched = true;
563  };
564  };
565  };
566  };
567  };
568 
569  var ret = new Array;
570 
571  for ( var c = 0; c < tmp.length; c++ )
572  {
573  if ( tmp[c].matches )
574  {
575  tmp[c].matches.sort ( sortMeetingSearchResponseCallback );
576  ret[ret.length] = tmp[c].matches;
577  };
578  };
579 
580  return ret;
581  };

583  /************************************************************************************//**
584  * \brief Callback used to sort the meeting response by weekday. *
585  * \returns 1 if a>b, -1 if a<b or 0 if they are equal. *
586  ****************************************************************************************/
587  function sortMeetingSearchResponseCallback ( in_mtg_obj_a, ///< Meeting object A
588  in_mtg_obj_b ///< Meeting Object B
589  )
590  {
591  var ret = 0;
592  var weekday_score_a = parseInt ( in_mtg_obj_a.weekday_tinyint, 10 );
593  var weekday_score_b = parseInt ( in_mtg_obj_b.weekday_tinyint, 10 );
594 
595  if ( weekday_score_a < g_Nouveau_start_week )
596  {
597  weekday_score_a += 7;
