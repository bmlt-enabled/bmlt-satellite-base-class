2 \file table_display.js *
3 \brief Javascript functions for the basic table display. *
4 * \
5 *
6 This file contains a function/object that implements the Fast Table Shortcode. Upon *
7 instantiation, it uses the page DOM to create a tabular display of meetings that are *
8 returned by a focused semantic search. The meetings are segregated by weekday, and only *
9 one weekday is populated at a time. Clicking on another weekday will start an AJAX *
10 query to the server, fetching the meetings for that weekday (and only that weekday). *
11 Search results are cached. *
12 *
13 The displayed meetings can be sorted by clicking on the column headers (with the sort *
14 direction being changed by repeated clicks). Changing the sort flushes all cached data. *
15 *
16 This class was designed to be highly self-contained. It can be instantiated multiple *
17 times per page, with each instance reflecting a different set of settings from the *
18 BMLT Administration setup (Theme, Week Start, Time Format and Root Server). The basic *
19 search parameters are passed in upon instantiation. These define the dataset to be *
20 fetched and displayed. The dataset is retrieved through a standard JSON semantic query *
21 to the instance's Root Server. *
22 *
23 The weekday tabs tend to be the "hubs" of the functionality. JSON search results are *
24 attached to the weekday tabs, and this data is used to create the weekday search *
25 display. The formats JSON is held by the main class, as that is used throughout the *
26 class. Otherwise, we try to encapsulate data and functionality as much as possible. *
27 *
28 This should only be used on Root Server version 2.7.8 or above, because the town sort *
29 will not work properly on older versions (but this will generally work, anyway). *
30 *
31 SOME NOTES ON CONVENTIONS: *
32 You will notice an abundance of semicolons. This is because this file is passed through *
33 an optimizer that crunches the entire file into one line. In this case, semicolons are *
34 absolutely vital. *
35 *
36 We use a lot of completion block functions, which is fairly standard for JavaScript. *
37 These are for things like mouseovers and click handlers. *
38 *
39 We use ul elements, as opposed to table elements, in order to maximize the potential *
40 for CSS modification of the displayed results (such as for responsive design). *

153/************************************************************************************//**
154 \brief A simple, generic AJAX request function.
155 \param url The URI to be called. This should include all parameters in GET form (even if this is POST).
156 \param callback The success callback
157 \param method The method ('get', 'GET', 'post' or 'POST')
158 \param extra_data If supplied, extra data to be delivered to the callback.
159 \returns a new XMLHTTPRequest object.
160 ****************************************************************************************/
161TableSearchDisplay.prototype.comms_AjaxRequest = function ( url, callback, method, extra_data ) {
162 /********************************************************************************//**
163 * \brief Create a generic XMLHTTPObject.
164 * This will account for the various flavors imposed by different browsers.
165 * \returns a new XMLHTTPRequest object.
166 ************************************************************************************/
167
168 function createXMLHTTPObject()
169 {
170 // Various systems will produce one or more instances of these classes.
171 var XMLHttpArray = [
172 function () {
173 return new XMLHttpRequest()}, // Your average browser.
174 function () {
175 return new ActiveXObject("Msxml2.XMLHTTP")}, // These are for Microsoft browsers.
176 function () {
177 return new ActiveXObject("Msxml2.XMLHTTP")},
178 function () {
179 return new ActiveXObject("Microsoft.XMLHTTP")}
180 ];
181
182 var xmlhttp = false;
183
184 // Go through the array until you hit a dispatcher that was properly instantiated.
185 // Once you find it, make that our dispatcher.
186 for ( var i=0; i < XMLHttpArray.length; i++ )
187 {
188 try
189 {
190 xmlhttp = XMLHttpArray[i]();
191 }
192 catch(e)
193 {
194 continue;
195 };
196 break;
197 };
198
199 return xmlhttp;
200 };
201
202 // Set up a dispatcher.
203 var req = createXMLHTTPObject();
204 req.finalCallback = callback; // Point to the given callback.
205 var sVars = null;
206 method = method.toString().toUpperCase(); // Clean the input method to all uppercase.
207
208 // Split the URL up, if this is a POST.
209 if (method == "POST" ) {
210 var rmatch = /^([^\?]*)\?(.*)$/.exec(url);
211 url = rmatch[1];
212 sVars = rmatch[2];
213 // This horrible, horrible kludge, is because Drupal insists on having its q parameter in the GET list only.
214 var rmatch_kludge = /(q=admin\/settings\/bmlt)&?(.*)/.exec(rmatch[2]);
215 if (rmatch_kludge && rmatch_kludge[1] ) {
216 url += '?'+rmatch_kludge[1];
217 sVars = rmatch_kludge[2];
218 };
219 };
220
221 // If we are passing some extra data to the callback, we do so here.
222 if (extra_data ) {
223 req.extra_data = extra_data;
224 };
225
226 // Create the HTTPRequest
227 req.open(method, url, true);
228
229 // In the case of POST, our data is separate from the URI.
230 if (method == "POST" ) {
231 req.setRequestHeader("Method", "POST "+url+" HTTP/1.1");
232 req.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
233 };
234
235 // Set the function to be called as the request does its thing.
236 req.onreadystatechange = function ( ) {
237 if (req.readyState != 4 ) { return; // We only care about the end.
238 }
239 if(req.status != 200 ) { return; // If we failed, we drop out.
240 }
241 callback(req, req.extra_data); // If we made it this far, we call our callback.
242 req = null; // Kill the request.
243 };
244
245 req.send(sVars); // Throw the switch, Igor!
246
247 return req;
248};

264 var tab_object = this.my_weekday_links[i];

266 };

268 // Ask for the formats used by the search set (not every format in the database).

272 this.m_ajax_request = this.comms_AjaxRequest(uri, this.comms_callback_loadloadFormatData, 'get', this);

336/************************************************************************************//**
337 \brief This turns military time to ante meridian format or cleaned military time.
338 If the time is midnight or noon, the localized string for that is returned.
339 \param in_time_string A string with the time in military format.
340 \returns A string, with the properly formatted time ("Noon", "Midnight", "HH:MM" or "HH:MM AM|PM").
341 ****************************************************************************************/
342TableSearchDisplay.prototype.utility_convertTime = function ( in_time_string ) {
343 var ret = in_time_string;
344 var time_array = in_time_string.split(":");
345 var hours = parseInt(time_array[0]);
346 var minutes = parseInt(time_array[1]);
347
348 // We use "Noon" and "Midnight" strings for those times.
349 // Kludge, to make "Midnight" late, as the computer wants it to be early.
350 // Anything over 23:54:59 is "Midnight."
351 if (((0 == hours) && (0 == minutes)) || ((hours == 23) && (minutes >= 55)) ) {
352 ret = g_table_ampm_array[3]; // Midnight
353 }
354 else if ((12 == hours) && (0 == minutes) ) {
355 ret = g_table_ampm_array[2]; // Noon
356 }
357 else if (!this.my_military_time ) {
358 var ampm = ( hours >= 12 ) ? g_table_ampm_array[1] : g_table_ampm_array[0];
359
360 // Subtract 12 from afternoon.
361 if (hours > 12 ) {
362 hours -= 12;
363 };
364
365 // Padding with a leading 0.
366 if (minutes < 10 ) {
367 minutes = "0" + minutes.toString();
368 };
369
370 ret = hours.toString() + ':' + minutes + ' ' + ampm;
371 }
372 else
373 {
374 // Padding with a leading 0.
375 if (minutes < 10 ) {
376 minutes = "0" + minutes.toString();
377 };
378
379
380 ret = hours.toString() + ':' + minutes.toString();
381 };
382
383 return ret;

506 function str_repeat( i,
507 m
508 ) {
509 for (var o = []; m > 0; o[--m] = i);
510 return o.join('');
511 };
512
513 var i = 0, a, f = arguments[i++], o = [], m, p, c, x, s = '';
514
515 while (f)
516 {
517 if (m = /^[^\x25]+/.exec(f)) {
518 o.push(m[0]);
519 }
520 else if (m = /^\x25{2}/.exec(f)) {
521 o.push('%');
522 }
523 else if (m = /^\x25(?:(\d+)\$)?(\+)?(0|'[^$])?(-)?(\d+)?(?:\.(\d+))?([b-fosuxX])/.exec(f)) {
524 if (((a = arguments[m[1] || i++]) == null) || (a == undefined)) {
525 throw('Too few arguments.');
526 };
527
528 if (/[^s]/.test(m[7]) && (typeof(a) != 'number')) {
529 throw('Expecting number but found ' + typeof(a));
530 };
531
532 switch (m[7])
533 {
534 case 'b': a = a.toString(2); break;
535 case 'c': a = String.fromCharCode(a); break;
536 case 'd': a = parseInt(a,10); break;
537 case 'e': a = m[6] ? a.toExponential(m[6]) : a.toExponential(); break;
538 case 'f': a = m[6] ? parseFloat(a).toFixed(m[6]) : parseFloat(a); break;
539 case 'o': a = a.toString(8); break;
540 case 's': a = ((a = String(a)) && m[6] ? a.substring(0, m[6]) : a); break;
541 case 'u': a = Math.abs(a); break;
542 case 'x': a = a.toString(16); break;
543 case 'X': a = a.toString(16).toUpperCase(); break;
544 };
545
546 a = (/[def]/.test(m[7]) && m[2] && a >= 0 ? '+' + a : a);
547 c = m[3] ? m[3] == '0' ? '0' : m[3].charAt(1) : ' ';
548 x = m[5] - String(a).length - s.length;
549 p = m[5] ? str_repeat(c, x) : '';
550 o.push(s + (m[4] ? a + p : p + a));
551 }
552 else
553 {
554 throw('Huh ?!');
555 };
556
557 f = f.substring(m[0].length);
558 };
559

561};
