313 /************************************************************************************//**
314  \brief This turns military time to ante meridian format or cleaned military time.
315  If the time is midnight or noon, the localized string for that is returned.
316  \param in_time_string A string with the time in military format.
317  \returns A string, with the properly formatted time ("Noon", "Midnight", "HH:MM" or "HH:MM AM|PM").
318 ****************************************************************************************/
319 BMLTQuickSearch.prototype.utility_convertTime = function ( in_time_string )
320 {
321  var ret = in_time_string;
322  var time_array = in_time_string.split ( ":" );
323  var hours = parseInt ( time_array[0] );
324  var minutes = parseInt ( time_array[1] );
325 
326  // We use "Noon" and "Midnight" strings for those times.
327  // Kludge, to make "Midnight" late, as the computer wants it to be early.
328  // Anything over 23:54:59 is "Midnight."
329  if ( ((0 == hours) && (0 == minutes)) || ((hours == 23) && (minutes >= 55)) )
330  {
331  ret = this.m_am_pm_array[3]; // Midnight
332  }
333  else if ( (12 == hours) && (0 == minutes) )
334  {
335  ret = this.m_am_pm_array[2]; // Noon
336  }
337  else if ( !this.m_military_time )
338  {
339  var ampm = ( hours >= 12 ) ? this.m_am_pm_array[1] : this.m_am_pm_array[0];
340 
341  // Subtract 12 from afternoon.
342  if ( hours > 12 )
343  {
344  hours -= 12;
345  };
346 
347  // Padding with a leading 0.
348  if ( minutes < 10 )
349  {
350  minutes = "0" + minutes.toString();
351  };
352 
353  ret = hours.toString() + ':' + minutes + ' ' + ampm;
354  }
355  else
356  {
357  // Padding with a leading 0.
358  if ( minutes < 10 )
359  {
360  minutes = "0" + minutes.toString();
361  };
362 
363 
364  ret = hours.toString() + ':' + minutes.toString();
365  };
366 
367  return ret;
368 };

545 * From here: https://stackoverflow.com/a/11076088/879365 *
546 ********************************************************************************************/
547 BMLTQuickSearch.prototype.intersectingArrays = function (/* pass all arrays here */) {
548  var output = [];
549  var cntObj = {};
550  var array, item, cnt;
551 
552  // for each array passed as an argument to the function
553  for (var i = 0; i < arguments.length; i++)
554  {
555  array = arguments[i];
556  // for each element in the array
557  for ( var j = 0; j < array.length; j++ )
558  {
559  item = "-" + array[j];
560  cnt = cntObj[item] || 0;
561  // if cnt is exactly the number of previous arrays,
562  // then increment by one so we count only one per array
563  if (cnt == i)
564  {
565  cntObj[item] = cnt + 1;
566  };
567  };
568  };
569 
570  // now collect all results that are in all arrays
571  for ( item in cntObj )
572  {
573  if (cntObj.hasOwnProperty(item) && cntObj[item] === arguments.length)
574  {
575  output.push(item.substring(1));
576  };
577  };
578 
579  return(output);
580 };

620 /****************************************************************************************//**
621 ********************************************************************************************/
622 BMLTQuickSearch.prototype.getSearchURI = function ( )
623 {
624  var value = null;
625  var ret = '';
626 
627  if ( this.m_values && this.m_weekday_objects )
628  {
629  // 1 value, the select is hidden.
630  if ( 1 == this.m_values.length )
631  {
632  value = this.m_values[0].value_name;
633  }
634  else
635  {
636  value = this.m_town_select.value;
637  };
638 
639  var valueIDs = [];
640 
641  for ( var index = 0; index < this.m_values.length; index++ )
642  {
643  var valueObject = this.m_values[index];
644 
645  var idArray = valueObject.ids.split(",");
646 
647  if ( valueObject.value_name == value )
648  {
649  valueIDs = idArray;
650  break;
651  }
652  else
653  {
654  if ( !value )
655  {
656  valueIDs.push.apply ( valueIDs, idArray );
657  };
658  };
659  };
660 
661  if ( valueIDs.length )
662  {
663  var resultArray = [];
664 
665  for ( var weekday = 1; weekday < 8; weekday++ )
666  {
667  var weekdayCheckbox = this.m_weekday_objects[weekday];
668  if ( weekdayCheckbox.checked )
669  {
670  var wValue = weekdayCheckbox.value;
671  for ( var index = 0; index < 7; index++ )
672  {
673  if ( this.m_weekdays[index].weekday_tinyint == wValue )
674  {
675  var idArray = this.m_weekdays[index].ids.split(",");
676 
677  if ( idArray && idArray.length )
678  {
679  if ( !resultArray || !resultArray.length )
680  {
681  resultArray = idArray;
682  }
683  else
684  {
685  resultArray.push.apply ( resultArray, idArray );
686  };
687  };
688  break;
689  };
690  };
691  };
692  };
693 
694  resultArray = this.intersectingArrays ( resultArray, valueIDs );
695 
696  if ( resultArray && resultArray.length )
697  {
698  ret = this.m_ajaxURI + "?bmlt_settings_id=" + this.m_optionsID + "&redirect_ajax_json=" + encodeURI ( 'switcher=GetSearchResults&get_used_formats=1' );
699 
700  if ( resultArray )
701  {
702  ret += encodeURI ( '&SearchString=' + resultArray.join(',') );
703  }
704  else
705  {
706  if ( !this.m_weekday_objects[0].checked )
707  {
708  var temp = [];
709  for ( var weekday = 1; weekday < 8; weekday++ )
710  {
711  var weekdayCheckbox = this.m_weekday_objects[weekday];
712  if ( weekdayCheckbox.checked )
713  {
714  temp.push ( 'weekdays[]=' + weekday.toString() );
715  };
716  };
717 
718  if ( temp.length )
719  {
720  ret += '&' + temp.join ( '&' );
721  }
722  else
723  {
724  ret = null;
725  };
726  };
727  };
728  };
729  };
730  };
731 
732  return ret;
733 };

772  {

774  };
775 
776  var b_weekday = parseInt ( b.weekday_tinyint ) - week_starts_on;
777  if ( b_weekday < 1 )
778  {
779  b_weekday += 7;
780  };
781 
782  var a_time = parseInt ( a.start_time.split ( ':' ).join('') );
783  var b_time = parseInt ( b.start_time.split ( ':' ).join('') );
784 
785  var a_town = a.location_municipality.trim().toLowerCase();
786  var b_town = b.location_municipality.trim().toLowerCase();
787 
788  if ( a_weekday < b_weekday )
789  {
790  return -1;
791  }
792  else
793  {
794  if ( b_weekday < a_weekday )
795  {
796  return 1;
797  }
798  else
799  {
800  if ( a_time < b_time )
801  {
802  return -1;
803  }
804  else
805  {
806  if ( b_time < a_time )
807  {
808  return 1;
809  }
810  else
811  {
812  if ( a_town < b_town )
813  {
814  return -1;
815  }
816  else
817  {
818  if ( b_town < a_town )
819  {
820  return 1;
821  };
822  };
823  };
824  };
825  };
826  };
827 
828  return 0;
829  });
830 
831  this.m_last_search_results_meetings = meetings;
832  this.m_last_search_results_formats = inSearchResults.formats;
833  };
834 
835  this.hideThrobber();
836  this.showResults();
