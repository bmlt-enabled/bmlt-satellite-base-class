297/************************************************************************************//**
298 \brief This turns military time to ante meridian format or cleaned military time.
299 If the time is midnight or noon, the localized string for that is returned.
300 \param in_time_string A string with the time in military format.
301 \returns A string, with the properly formatted time ("Noon", "Midnight", "HH:MM" or "HH:MM AM|PM").
302 ****************************************************************************************/
303BMLTQuickSearch.prototype.utility_convertTime = function ( in_time_string ) {
304 var ret = in_time_string;
305 var time_array = in_time_string.split(":");
306 var hours = parseInt(time_array[0]);
307 var minutes = parseInt(time_array[1]);
308
309 // We use "Noon" and "Midnight" strings for those times.
310 // Kludge, to make "Midnight" late, as the computer wants it to be early.
311 // Anything over 23:54:59 is "Midnight."
312 if (((0 == hours) && (0 == minutes)) || ((hours == 23) && (minutes >= 55)) ) {
313 ret = this.m_am_pm_array[3]; // Midnight
314 }
315 else if ((12 == hours) && (0 == minutes) ) {
316 ret = this.m_am_pm_array[2]; // Noon
317 }
318 else if (!this.m_military_time ) {
319 var ampm = ( hours >= 12 ) ? this.m_am_pm_array[1] : this.m_am_pm_array[0];
320
321 // Subtract 12 from afternoon.
322 if (hours > 12 ) {
323 hours -= 12;
324 };
325
326 // Padding with a leading 0.
327 if (minutes < 10 ) {
328 minutes = "0" + minutes.toString();
329 };
330
331 ret = hours.toString() + ':' + minutes + ' ' + ampm;
332 }
333 else
334 {
335 // Padding with a leading 0.
336 if (minutes < 10 ) {
337 minutes = "0" + minutes.toString();
338 };
339
340
341 ret = hours.toString() + ':' + minutes.toString();
342 };
343
344 return ret;
345};

397 ret += '<span class="bmlt_quicksearch_town_comma_span">, </span>';

405 ret += in_json_data.location_province.toString();
406 ret += '</span>';
407 };
408
409 if (hasTown || (in_json_data.location_province && in_json_data.location_province.toString()) && (in_json_data.location_postal_code_1 && in_json_data.location_postal_code_1.toString()) ) {
410 ret += '<span class="bmlt_quicksearch_zip_space_span"> </span>';

416
417 ret += in_json_data.location_postal_code_1.toString();
418 ret += '</span>';

502/****************************************************************************************//**
503 * From here: https://stackoverflow.com/a/11076088/879365 *
504 ********************************************************************************************/
505BMLTQuickSearch.prototype.intersectingArrays = function (/* pass all arrays here */) {
506 var output = [];
507 var cntObj = {};
508 var array, item, cnt;
509
510 // for each array passed as an argument to the function
511 for (var i = 0; i < arguments.length; i++)
512 {
513 array = arguments[i];
514 // for each element in the array
515 for ( var j = 0; j < array.length; j++ )
516 {
517 item = "-" + array[j];
518 cnt = cntObj[item] || 0;
519 // if cnt is exactly the number of previous arrays,
520 // then increment by one so we count only one per array
521 if (cnt == i) {
522 cntObj[item] = cnt + 1;
523 };
524 };
525 };
526
527 // now collect all results that are in all arrays
528 for ( item in cntObj )
529 {
530 if (cntObj.hasOwnProperty(item) && cntObj[item] === arguments.length) {
531 output.push(item.substring(1));
532 };
533 };
534
535 return(output);
536};
