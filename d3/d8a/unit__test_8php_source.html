49 /****************************************************************************************//**
50 * \brief Gets a cleaned base server path (accounting for SSL and non-standard ports). *
51 * *
52 * \returns A string. The Server URI. *
53 ********************************************************************************************/
54 function getCleanBaseURI()
55 {
56  $port = $_SERVER['SERVER_PORT'] ;
57  // IIS puts "off" in the HTTPS field, so we need to test for that.
58  $https = (!empty($_SERVER['HTTPS']) && (($_SERVER['HTTPS'] !== 'off') || ($port == 443)));
59  $server_path = $_SERVER['SERVER_NAME'];
60  $my_path = $_SERVER['PHP_SELF'];
61  $server_path .= trim((($https && ($port != 443)) || (!$https && ($port != 80))) ? ':'.$port : '', '/');
62  $server_path = 'http'.($https ? 's' : '').'://'.$server_path.$my_path;
63 
64  return $server_path;
65 }

110 /****************************************************************************************//**
111 * \brief Decides which operation to perform, based on the utest_string parameter. *
112 * *
113 * \returns a string. Either 'admin', 'render' or null. *
114 ********************************************************************************************/
115 function u_test_operation()
116 {
117  $ret = null;
118 
119  global $BMLTPluginOp;
120 
121  $oper_text = u_test_get_string();
122 
123  if (strtolower($oper_text) == 'admin') {
124  $ret = 'admin';
125  } elseif (strtolower($oper_text) == 'clear_session' || (strtolower($oper_text) == 'clear_session_start')) {
126  $ret = 'clear_session';
127  $firsts = $BMLTPluginOp->getBMLTOptions(1);
128  $firsts['setting_name'] = 'GNYR';
129  $firsts['id'] = '1';
130  $BMLTPluginOp->setBMLTOptions($firsts, 1);
131  } elseif ($BMLTPluginOp->get_shortcode('bmlt', $oper_text)) {
132  $ret = 'render-old';
133  } elseif (isset($oper_text) || count($_GET) || count($_POST)) {
134  $ret = 'render-new';
135  }
136 
137  return $ret;
138 }
